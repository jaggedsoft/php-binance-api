881  private function httpRequest(string $url, string $method = "GET", array $params = [], bool $signed = false)
882  {
883  if (function_exists('curl_init') == false) {
884  throw new \Exception("Sorry cURL is not installed!");
885  }
886 
887  $ch = curl_init();
888  curl_setopt($ch, CURLOPT_VERBOSE, $this->httpDebug);
889  $query = http_build_query($params, '', '&');
890 
891  // signed with params
892  if ($signed == true) {
893  if (empty($this->api_key)) {
894  throw new \Exception("signedRequest error: API Key not set!");
895  }
896 
897  if (empty($this->api_secret)) {
898  throw new \Exception("signedRequest error: API Secret not set!");
899  }
900 
901  $base = $this->base;
902  $ts = (microtime(true) * 1000) + $this->info['timeOffset'];
903  $params['timestamp'] = number_format($ts, 0, '.', '');
904  if (isset($params['wapi'])) {
905  unset($params['wapi']);
906  $base = $this->wapi;
907  }
908  $query = http_build_query($params, '', '&');
909  $signature = hash_hmac('sha256', $query, $this->api_secret);
910  $endpoint = $base . $url . '?' . $query . '&signature=' . $signature;
911  curl_setopt($ch, CURLOPT_URL, $endpoint);
912  curl_setopt($ch, CURLOPT_HTTPHEADER, array(
913  'X-MBX-APIKEY: ' . $this->api_key,
914  ));
915  }
916  // params so buildquery string and append to url
917  else if (count($params) > 0) {
918  curl_setopt($ch, CURLOPT_URL, $this->base . $url . '?' . $query);
919  }
920  // no params so just the base url
921  else {
922  curl_setopt($ch, CURLOPT_URL, $this->base . $url);
923  curl_setopt($ch, CURLOPT_HTTPHEADER, array(
924  'X-MBX-APIKEY: ' . $this->api_key,
925  ));
926  }
927  curl_setopt($ch, CURLOPT_USERAGENT, "User-Agent: Mozilla/4.0 (compatible; PHP Binance API)");
928  // Post and postfields
929  if ($method == "POST") {
930  curl_setopt($ch, CURLOPT_POST, true);
931  // curl_setopt($ch, CURLOPT_POSTFIELDS, $query);
932  }
933  // Delete Method
934  if ($method == "DELETE") {
935  curl_setopt($ch, CURLOPT_CUSTOMREQUEST, $method);
936  }
937  // proxy settings
938  if (is_array($this->proxyConf)) {
939  curl_setopt($ch, CURLOPT_PROXY, $this->getProxyUriString());
940  if (isset($this->proxyConf['user']) && isset($this->proxyConf['pass'])) {
941  curl_setopt($ch, CURLOPT_PROXYUSERPWD, $this->proxyConf['user'] . ':' . $this->proxyConf['pass']);
942  }
943  }
944  curl_setopt($ch, CURLOPT_FOLLOWLOCATION, true);
945  // headers will proceed the output, json_decode will fail below
946  curl_setopt($ch, CURLOPT_HEADER, false);
947  curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
948  curl_setopt($ch, CURLOPT_TIMEOUT, 60);
949 
950  // set user defined curl opts last for overriding
951  foreach ($this->curlOpts as $key => $value) {
952  curl_setopt($ch, constant($key), $value);
953  }
954 
955  $output = curl_exec($ch);
956  // Check if any error occurred
957  if (curl_errno($ch) > 0) {
958  echo 'Curl error: ' . curl_error($ch) . "\n";
959  return [];
960  }
961  curl_close($ch);
962  $json = json_decode($output, true);
963  if (isset($json['msg'])) {
964  echo "signedRequest error: {$output}" . PHP_EOL;
965  }
966  $this->transfered += strlen($output);
967  $this->requestCount++;
968  return $json;
969  }
